# 1. Create the product tuple
product = ("Laptop", 50000, "Black", "Samsung", "Electronics")
print("Product tuple:", product)

# 2. Access and print the second element
print("Second element:", product[1])

# 3. Slice and print the last two elements
print("Last two elements:", product[-2:])

# 4. Check for "Electronics" in the tuple
if "Electronics" in product:
    print('"Electronics" is present in the product tuple.')

# 5. Create a tuple of prices and count 1000
prices = (1000, 1500, 1200, 1100, 900)
print("Count of 1000 in prices:", prices.count(1000))

# 6. Find and print max and min price
print("Maximum price:", max(prices))
print("Minimum price:", min(prices))

# 7. Loop to print each item in product tuple
print("Items in product tuple:")
for item in product:
    print(item)

# 8. Convert to list, change price, convert back
product_list = list(product)
product_list[1] = 55000
product = tuple(product_list)
print("Updated product tuple:", product)

# 9. Add "In Stock" by concatenation
product = product + ("In Stock",)
print("Product tuple after adding 'In Stock':", product)

# 10. Remove "Electronics"
product_list = list(product)
product_list.remove("Electronics")
product = tuple(product_list)
print("Product tuple after removing 'Electronics':", product)

# 11. Unpack into three variables (name, price, rest)
name, price, *rest = product
print("Name:", name)
print("Price:", price)
print("Other details:", rest)

# 12. Create nested tuple and access second product's name
product1 = ("Laptop", 50000, "Black", "Samsung", "Electronics")
product2 = ("Phone", 30000, "White", "Apple", "Electronics")
product3 = ("Tablet", 20000, "Gray", "Lenovo", "Electronics")
nested_products = (product1, product2, product3)
print("Name of second product:", nested_products[1][0])
